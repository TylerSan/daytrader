```python
"""Phase 2 decision gate — spec §3.1 machine-readable table walker.

Loads data source CSVs from a directory, walks the 6-row decision table
top-down (first match wins), and returns the matching row + family.
Row 6 ("null") is the fallthrough.

Pre-committed: thresholds and priority are frozen at git tag `phase1-frozen`.
Any later change requires explicit spec amendment + new commit + re-tag.

Usage:
    .venv/bin/python scripts/post_w2_phase2_decide.py docs/research/post-w2/
"""

from __future__ import annotations

import json
import sys
from pathlib import Path
from typing import Optional

import pandas as pd


# Thresholds — FROZEN by spec §3.3.
ROW1_SHARPE = 1.0      # MES S1 pure OOS
ROW2_SHARPE = 1.0      # D3 mean reversion pure OOS
ROW3_SHARPE = 1.0      # D5 multi-instrument max
ROW4_SHARPE = 1.0      # D4 gap pure OOS
ROW5_SHARPE = 1.0      # D6 event on-event subset
ROW5_N_MIN = 30        # D6 event minimum trade count


def _safe_read(path: Path) -> Optional[pd.DataFrame]:
    """Return DataFrame if file exists, else None. Missing = row skipped."""
    if not path.exists():
        return None
    return pd.read_csv(path)


def _row1_check(csv_dir: Path) -> Optional[dict]:
    """MES S1 pure OOS — family was right, instrument was wrong."""
    df = _safe_read(csv_dir / "mes_b3_report.csv")
    if df is None:
        return None
    max_sharpe = df["sharpe_pure_oos_gross"].max()
    if max_sharpe >= ROW1_SHARPE:
        return {
            "row": 1,
            "family": "B_prime_mes_native_trend",
            "metric": "mes_b3_report.sharpe_pure_oos_gross (max over S1a, S1b)",
            "observed": float(max_sharpe),
            "threshold": ROW1_SHARPE,
            "action": "Start Phase 3 spec: MES S1 bake-off under W2 discipline",
        }
    return None


def _row2_check(csv_dir: Path) -> Optional[dict]:
    """D3 mean reversion baseline on SPY pure OOS."""
    df = _safe_read(csv_dir / "d3_mr_summary.csv")
    if df is None:
        return None
    row = df[df["window"] == "pure_oos"]
    if row.empty:
        return None
    sharpe = float(row["sharpe_gross"].iloc[0])
    if sharpe >= ROW2_SHARPE:
        return {
            "row": 2,
            "family": "C_mean_reversion",
            "metric": "d3_mr_summary.sharpe_gross where window=pure_oos",
            "observed": sharpe,
            "threshold": ROW2_SHARPE,
            "action": "Start Phase 3 spec: SPY mean-reversion bake-off",
        }
    return None


def _row3_check(csv_dir: Path) -> Optional[dict]:
    """D5 multi-instrument max Sharpe over 4 (QQQ|IWM × S1a|S1b)."""
    df = _safe_read(csv_dir / "d5_compare.csv")
    if df is None:
        return None
    max_sharpe = df["sharpe_gross"].max()
    if max_sharpe >= ROW3_SHARPE:
        return {
            "row": 3,
            "family": "B_multi_instrument_trend",
            "metric": "d5_compare.sharpe_gross (max over 4 rows)",
            "observed": float(max_sharpe),
            "threshold": ROW3_SHARPE,
            "action": "Start Phase 3 spec: multi-ETF S1 bake-off",
        }
    return None


def _row4_check(csv_dir: Path) -> Optional[dict]:
    """D4 overnight gap-fill on SPY pure OOS."""
    df = _safe_read(csv_dir / "d4_gap_summary.csv")
    if df is None:
        return None
    row = df[df["window"] == "pure_oos"]
    if row.empty:
        return None
    sharpe = float(row["sharpe_gross"].iloc[0])
    if sharpe >= ROW4_SHARPE:
        return {
            "row": 4,
            "family": "overnight_gap",
            "metric": "d4_gap_summary.sharpe_gross where window=pure_oos",
            "observed": sharpe,
            "threshold": ROW4_SHARPE,
            "action": "Start Phase 3 spec: SPY gap-fill bake-off",
        }
    return None


def _row5_check(csv_dir: Path) -> Optional[dict]:
    """D6 on-event subset — needs BOTH Sharpe and n gates."""
    df = _safe_read(csv_dir / "d6_events_summary.csv")
    if df is None:
        return None
    row = df[df["partition"] == "on_event"]
    if row.empty:
        return None
    sharpe = float(row["sharpe_gross"].iloc[0])
    n = int(row["n"].iloc[0])
    if sharpe >= ROW5_SHARPE and n >= ROW5_N_MIN:
        return {
            "row": 5,
            "family": "event_conditional",
            "metric": "d6_events_summary: sharpe_gross AND n where partition=on_event",
            "observed": {"sharpe_gross": sharpe, "n": n},
            "threshold": {"sharpe_gross": ROW5_SHARPE, "n_min": ROW5_N_MIN},
            "action": "Start Phase 3 spec: event-day subset bake-off",
        }
    return None


def _null_row() -> dict:
    """Row 6 — no family passes. Write null memo, close mechanical track."""
    return {
        "row": 6,
        "family": "null",
        "action": (
            "Write null memo. Formally close mechanical research branch. "
            "User moves to W2 handoff (b) disciplined discretionary or "
            "(c) pause. Any future family attempt requires written new "
            "evidence per spec R7."
        ),
    }


def decide(csv_dir: Path) -> dict:
    """Walk spec §3.1 decision table; first match wins."""
    csv_dir = Path(csv_dir)
    for check in (_row1_check, _row2_check, _row3_check, _row4_check, _row5_check):
        result = check(csv_dir)
        if result is not None:
            return result
    return _null_row()


def main():
    if len(sys.argv) != 2:
        print("Usage: post_w2_phase2_decide.py <csv_dir>", file=sys.stderr)
        sys.exit(1)
    csv_dir = Path(sys.argv[1])
    if not csv_dir.exists():
        print(f"ERROR: {csv_dir} does not exist", file=sys.stderr)
        sys.exit(1)
    result = decide(csv_dir)
    print(json.dumps(result, indent=2, default=str))


if __name__ == "__main__":
    main()
```

### - [ ] Step 1: Implement pre-flight script

Create `scripts/mes_b1_preflight.py`:

```python
"""B1 — Databento MES pre-flight cost check (spec §2 B1).

Estimates the cost of fetching GLBX.MDP3 MES 1m continuous c-1 for the
full window before committing to a paid pull. Applies spec §B1 gate:

    <= $20    → proceed with full window 2018-05-01 → 2024-12-31
    $20-50    → downgrade window to 3y (2022-01-01 → 2024-12-31) and re-check
    > $50     → abort B-line, decision row 1 skipped

Usage:
    DATABENTO_API_KEY=<key> .venv/bin/python scripts/mes_b1_preflight.py
"""

from __future__ import annotations

import os
import sys
from pathlib import Path

import databento

FULL_START = "2018-05-01"
FULL_END = "2024-12-31"
DOWNGRADE_START = "2022-01-01"
DOWNGRADE_END = "2024-12-31"

GATE_PROCEED = 20.0
GATE_ABORT = 50.0

OUTPUT_PATH = Path("docs/research/post-w2/mes_preflight.txt")


def estimate(api_key: str, start: str, end: str) -> float:
    client = databento.Historical(api_key)
    return float(client.metadata.get_cost(
        dataset="GLBX.MDP3",
        schema="ohlcv-1m",
        symbols=["MES.c.0"],
        stype_in="continuous",
        start=start,
        end=end,
    ))


def main():
    api_key = os.environ.get("DATABENTO_API_KEY")
    if not api_key:
        print("ERROR: DATABENTO_API_KEY not set", file=sys.stderr)
        sys.exit(1)

    full_cost = estimate(api_key, FULL_START, FULL_END)
    lines = [
        f"Window: {FULL_START} → {FULL_END}",
        f"Dataset: GLBX.MDP3, schema=ohlcv-1m, symbol=MES.c.0 (continuous c-1)",
        f"Cost estimate: ${full_cost:.4f}",
    ]

    if full_cost <= GATE_PROCEED:
        decision = "PROCEED — within ≤ $20 budget"
        final_start, final_end = FULL_START, FULL_END
    elif full_cost <= GATE_ABORT:
        down_cost = estimate(api_key, DOWNGRADE_START, DOWNGRADE_END)
        lines.append("")
        lines.append(f"Downgrade to 3y window: {DOWNGRADE_START} → {DOWNGRADE_END}")
```

```python
# Invariants.
# 6.5y * ~252 trading days * 390 RTH bars = ~640k, tolerate ± 10%
BARS_EXPECTED = 640_000
BARS_TOLERANCE = 0.10
# ~26 quarterly rollovers * 2 days skip = ~52, tolerate ± 10 absolute
ROLLOVER_EXPECTED = 52
ROLLOVER_TOLERANCE_ABS = 10
# Low coverage days should be < 3% of total days
LOW_COVERAGE_THRESHOLD_PCT = 3.0


def check(start: date, end: date, api_key: str) -> tuple[bool, list[str]]:
    years = (end - start).days / 365.25
    bars_expected = int(BARS_EXPECTED * years / 6.5)

    ds = load_mes_1m(start=start, end=end, api_key=api_key, cache_dir=CACHE_DIR)

    lines = [
        f"# B2 MES Loader Health Check",
        "",
        f"**Window:** {start} → {end} ({years:.2f}y)",
        "",
        "| Invariant | Expected | Actual | Status |",
        "|---|---|---|---|",
    ]
    all_ok = True

    # 1. Bar count
    n_bars = len(ds.bars)
    low = int(bars_expected * (1 - BARS_TOLERANCE))
    high = int(bars_expected * (1 + BARS_TOLERANCE))
    ok1 = low <= n_bars <= high
    all_ok &= ok1
    lines.append(f"| bars row count | {low}–{high} | {n_bars} | {'PASS' if ok1 else 'FAIL'} |")

    # 2. Rollover count
    n_roll = len(ds.rollover_skip_dates)
    roll_low = int(ROLLOVER_EXPECTED * years / 6.5) - ROLLOVER_TOLERANCE_ABS
    roll_high = int(ROLLOVER_EXPECTED * years / 6.5) + ROLLOVER_TOLERANCE_ABS
    ok2 = roll_low <= n_roll <= roll_high
    all_ok &= ok2
    lines.append(f"| rollover_skip_dates | {roll_low}–{roll_high} | {n_roll} | {'PASS' if ok2 else 'FAIL'} |")

    # 3. Low-coverage rate
    qa = ds.quality_report
    n_total = len(qa)
    n_low = int(qa["flag_low_coverage"].sum())
    pct_low = (n_low / n_total * 100.0) if n_total > 0 else 0.0
    ok3 = pct_low < LOW_COVERAGE_THRESHOLD_PCT
    all_ok &= ok3
    lines.append(f"| low-coverage days (<99% bars) | <{LOW_COVERAGE_THRESHOLD_PCT}% | {pct_low:.2f}% ({n_low}/{n_total}) | {'PASS' if ok3 else 'FAIL'} |")

    lines.append("")
    lines.append(f"**Overall: {'PASS' if all_ok else 'FAIL'}**")
    if not all_ok:
        lines.append("")
        lines.append("B3 must not run until FAIL is resolved. See spec §2 B2.")

    return all_ok, lines
```

```python
def _apply_costs(trades):
    """Per-trade net PnL series in USD using legacy MES round_trip_cost_usd()."""
    return pd.Series([
        (t.exit_price - t.entry_price if t.direction == "long" else t.entry_price - t.exit_price)
        * MES_POINT_VALUE
        - round_trip_cost_usd(contracts=1, exit_kind=t.outcome.value)
        for t in trades
    ])


def _evaluate_window(trades, label):
    """Compute metrics for a trade subset."""
    if len(trades) == 0:
        return {"window": label, "n_trades": 0}
    gross = pd.Series([
        (t.exit_price - t.entry_price if t.direction == "long" else t.entry_price - t.exit_price)
        * MES_POINT_VALUE
        for t in trades
    ])
    net = _apply_costs(trades)
    trade_dates = [_date.fromisoformat(t.date) for t in trades]
    eq = equity_curve_from_pnl(net, starting_capital=STARTING_CAPITAL)
    dr_net = daily_returns_from_pnl(net, trade_dates, starting_capital=STARTING_CAPITAL)
    dr_gross = daily_returns_from_pnl(gross, trade_dates, starting_capital=STARTING_CAPITAL)

    return {
        "window": label,
        "n_trades": len(trades),
        f"sharpe_{label}_gross": annualized_sharpe(dr_gross),
        f"sharpe_{label}_net": annualized_sharpe(dr_net),
        "sortino_net": annualized_sortino(dr_net),
        "calmar_net": calmar_ratio(eq, trading_days=252),
        "max_dd": max_drawdown(eq),
        "longest_dd_days": longest_drawdown_duration(eq),
        "profit_factor": profit_factor(net),
        "expectancy_r": expectancy_r(pd.Series([t.r_multiple for t in trades])),
        "dsr_pvalue": deflated_sharpe_pvalue(dr_net, n_trials=N_TRIALS),
        "total_net_pnl": float(net.sum()),
    }


def run_s1_on_mes(
    bars,
    rollover_skip_dates,
    low_coverage_dates,
    replication_start,
    replication_end,
    pure_oos_start,
    pure_oos_end,
    output_dir,
):

    skip = set(rollover_skip_dates) | set(low_coverage_dates)
    # Filter bars by date skip (intersection of local ET date and skip set)
    from zoneinfo import ZoneInfo
    et = ZoneInfo("America/New_York")
    local_dates = bars.index.tz_convert(et).date
    mask = ~pd.Series(local_dates, index=bars.index).isin(skip)
    clean = bars[mask]

    output_dir = Path(output_dir)
    output_dir.mkdir(parents=True, exist_ok=True)
    rows = []

    for variant_name, cls in [("s1a", S1a_ORB_TargetAndEOD), ("s1b", S1b_ORB_EODOnly)]:
        strat = cls(symbol="MES", point_value=MES_POINT_VALUE)
        trades = strat.run(clean)

        # Save trade log
        trade_df = pd.DataFrame([{
            **asdict(t),
            "outcome": t.outcome.value,
        } for t in trades])
        trade_df.to_csv(output_dir / f"mes_b3_{variant_name}.csv", index=False)

        # Evaluate each window
        repl_trades = filter_trades_by_window(trades, replication_start, replication_end)
        oos_trades = filter_trades_by_window(trades, pure_oos_start, pure_oos_end)

        rep_metrics = _evaluate_window(repl_trades, "replication")
        oos_metrics = _evaluate_window(oos_trades, "pure_oos")

        # Flatten into single row per variant
        rows.append({
            "variant": variant_name,
            "n_trades_replication": rep_metrics.get("n_trades", 0),
            "n_trades_pure_oos": oos_metrics.get("n_trades", 0),
            "sharpe_pure_oos_gross": oos_metrics.get("sharpe_pure_oos_gross", float("nan")),
            "sharpe_pure_oos_net": oos_metrics.get("sharpe_pure_oos_net", float("nan")),
```

```python
Joins Plan 3 S1a/S1b trade log to daily features:
  - overnight gap direction (up/down/none)
  - prev-day range tertile
  - day-of-week

Usage:
    .venv/bin/python scripts/diagnose_d1_per_day.py
"""

from __future__ import annotations

from pathlib import Path

import numpy as np
import pandas as pd

PLAN3_TRADES = Path("docs/research/bakeoff/plan3_main_report.csv")
SPY_DAILY_CACHE = Path("data/cache/ohlcv/SPY_1d_2018-05-01_2024-12-31_raw.parquet")

OUT_CSV = Path("docs/research/post-w2/d1_per_day.csv")
OUT_MD = Path("docs/research/post-w2/d1_feature_table.md")

GAP_THRESHOLD = 0.001   # 0.1%


def classify_gap(gap_pct: float) -> str:
    if gap_pct > GAP_THRESHOLD:
        return "up"
    if gap_pct < -GAP_THRESHOLD:
        return "down"
    return "none"


def aggregate_by_feature(df: pd.DataFrame, feature_col: str) -> pd.DataFrame:
    """Return bucket × {n, win_rate, mean_pnl}."""
    rows = []
    for bucket, g in df.groupby(feature_col):
        rows.append({
            "feature": feature_col,
            "bucket": bucket,
            "n": len(g),
            "win_rate": float(g["won"].mean()),
            "mean_pnl": float(g["pnl"].mean()),
        })
    return pd.DataFrame(rows)


def _build_day_features(spy_daily: pd.DataFrame, events: pd.DataFrame) -> pd.DataFrame:
    """Given SPY daily OHLC and event calendar, emit per-day feature rows."""
    d = spy_daily.copy()
    d["gap_pct"] = (d["open"] - d["close"].shift(1)) / d["close"].shift(1)
    d["gap_class"] = d["gap_pct"].map(classify_gap, na_action="ignore")
    d["prev_range"] = (d["high"].shift(1) - d["low"].shift(1)) / d["close"].shift(1)
    d["prev_range_tertile"] = pd.qcut(
        d["prev_range"], 3, labels=["low", "mid", "high"]
    ).astype(str)
    d["day_of_week"] = pd.to_datetime(d.index).day_name()
    d["is_fomc"] = d.index.isin(
        events[events["event_type"] == "fomc"]["date"].values
    )
    d["is_cpi"] = d.index.isin(
        events[events["event_type"] == "cpi"]["date"].values
    )
    d["is_nfp"] = d.index.isin(
        events[events["event_type"] == "nfp"]["date"].values
    )
    d["is_any_event"] = d[["is_fomc", "is_cpi", "is_nfp"]].any(axis=1)
    return d


def main():
    trades = pd.read_csv(PLAN3_TRADES)
    # Plan 3 CSV is aggregated metrics, not per-trade. Per-trade logs are
    # plan3_s1a.csv / plan3_s1b.csv (existing Plan 3 outputs). Use those.
    s1a_path = PLAN3_TRADES.parent / "plan3_s1a.csv"
    s1b_path = PLAN3_TRADES.parent / "plan3_s1b.csv"
    if not (s1a_path.exists() and s1b_path.exists()):
        print(f"ERROR: per-trade logs not found ({s1a_path}, {s1b_path})",
              file=__import__('sys').stderr)
        __import__('sys').exit(1)

    s1a = pd.read_csv(s1a_path)
    s1a["variant"] = "s1a"
    s1b = pd.read_csv(s1b_path)
    s1b["variant"] = "s1b"
    all_trades = pd.concat([s1a, s1b], ignore_index=True)
    all_trades["trade_date"] = pd.to_datetime(all_trades["date"]).dt.date
    all_trades["won"] = all_trades["exit_price"] > all_trades["entry_price"]
    all_trades.loc[all_trades["direction"] == "short", "won"] = (
        all_trades.loc[all_trades["direction"] == "short", "exit_price"]
        < all_trades.loc[all_trades["direction"] == "short", "entry_price"]
    )
    all_trades["pnl"] = np.where(
        all_trades["direction"] == "long",
        all_trades["exit_price"] - all_trades["entry_price"],
        all_trades["entry_price"] - all_trades["exit_price"],
    )

    # Daily features
    if not SPY_DAILY_CACHE.exists():
        # Fall back: skip prev_range + gap if cache missing
        print(f"WARN: {SPY_DAILY_CACHE} missing; prev_range + gap_class dropped",
              file=__import__('sys').stderr)
        day_feat = None
    else:
        spy_daily = pd.read_parquet(SPY_DAILY_CACHE)
        if "SPY" in spy_daily.columns.get_level_values(0):
            spy_daily = spy_daily["SPY"]
        events = pd.read_csv(EVENT_CAL)
        events["date"] = pd.to_datetime(events["date"]).dt.date
        day_feat = _build_day_features(spy_daily, events)
        day_feat = day_feat.reset_index().rename(columns={"index": "trade_date"})
        day_feat["trade_date"] = pd.to_datetime(day_feat["trade_date"]).dt.date

    if day_feat is not None:
        joined = all_trades.merge(
            day_feat[["trade_date", "gap_class", "prev_range_tertile",
                      "day_of_week", "is_fomc", "is_cpi", "is_nfp", "is_any_event"]],
            on="trade_date", how="left",
        )
    else:
        joined = all_trades.assign(
            gap_class="unknown", prev_range_tertile="unknown",
            day_of_week="unknown", is_fomc=False, is_cpi=False,
            is_nfp=False, is_any_event=False,
        )

    OUT_CSV.parent.mkdir(parents=True, exist_ok=True)
    joined.to_csv(OUT_CSV, index=False)

    # Aggregation markdown
    features = ["gap_class", "prev_range_tertile", "day_of_week", "is_any_event"]

    OUT_MD.write_text("\n".join(md) + "\n")
    print(OUT_MD.read_text())


if __name__ == "__main__":
    main()
```

```python
"""D2 — Signal autocorrelation on SPY 1-minute bars (spec §2 D2).

Computes lag-1/5/15 autocorrelation of forward returns, bucketed by ET hour.

Usage:
    .venv/bin/python scripts/diagnose_d2_signal_autocorr.py
"""

from __future__ import annotations

from pathlib import Path

import numpy as np
import pandas as pd

SPY_CACHE = Path("data/cache/ohlcv/SPY_1m_2018-05-01_2024-12-31_raw.parquet")
OUT_CSV = Path("docs/research/post-w2/d2_signal.csv")
OUT_MD = Path("docs/research/post-w2/d2_autocorr.md")

LAGS = (1, 5, 15)


def autocorr_by_hour(df: pd.DataFrame, col: str, lag: int) -> pd.DataFrame:
    """Hour-bucketed autocorrelation for one lag."""
    rows = []
    for hour, g in df.groupby(df.index.hour):
        s = g[col].dropna()
        if len(s) < lag + 2:
            continue
        ac = float(s.autocorr(lag=lag))
        rows.append({"hour_et": int(hour), "lag": lag, "autocorr": ac, "n": len(s)})
    return pd.DataFrame(rows)


def main():
    if not SPY_CACHE.exists():
        print(f"ERROR: {SPY_CACHE} missing; run SPY loader first", file=__import__('sys').stderr)
        __import__('sys').exit(1)

    df = pd.read_parquet(SPY_CACHE)
    from zoneinfo import ZoneInfo
    et = ZoneInfo("America/New_York")
    df.index = df.index.tz_convert(et)

    # RTH only (09:30-15:59)
    mask = (df.index.time >= pd.Timestamp("09:30").time()) & (df.index.time < pd.Timestamp("16:00").time())
    df = df[mask]

    close = df["close"]
    for n in (1, 5, 15):
        df[f"fwd_ret_{n}"] = (close.shift(-n) - close) / close

    OUT_CSV.parent.mkdir(parents=True, exist_ok=True)
    df[[f"fwd_ret_{n}" for n in (1, 5, 15)]].to_csv(OUT_CSV)

    chunks = []
    for lag in LAGS:
        chunks.append(autocorr_by_hour(df, "fwd_ret_1", lag))
    agg = pd.concat(chunks, ignore_index=True)

    md = ["# D2 Signal Autocorrelation — SPY 1m forward-return", "",
          "Hour-bucketed autocorrelation of 1-bar forward return at lags 1/5/15.",
          "",
          agg.to_markdown(index=False, floatfmt=".4f"),
          "",
          "**Reading:** |autocorr| > 0.05 across many buckets → mean-reversion or "
          "trend structure worth exploring; |autocorr| < 0.01 across → near-white-noise."]
    OUT_MD.write_text("\n".join(md) + "\n")
    print(OUT_MD.read_text())


if __name__ == "__main__":
    main()
```

```python
"""D3 — Naive mean-reversion baseline on SPY 1m (spec §2 D3).

Rules (intentionally simple, no tuning):
  - At 09:35 ET: if close[09:34] > open[09:30] → short;
                  if close[09:34] < open[09:30] → long;
                  if equal → skip
  - Entry price = close[09:35]
  - Exit earlier of:
      (a) price returns to cumulative VWAP (long: price ≥ vwap; short: price ≤ vwap)
      (b) 15:55 ET forced EOD flat
  - No stop, no costs (gross)

Decision gate: row 2 — pure_oos sharpe_gross ≥ 1.0

Usage:
    .venv/bin/python scripts/diagnose_d3_meanrev_baseline.py
"""

from __future__ import annotations

from datetime import date, time, timedelta
from pathlib import Path
from zoneinfo import ZoneInfo

import numpy as np
import pandas as pd

from daytrader.research.bakeoff.metrics import annualized_sharpe

SPY_CACHE = Path("data/cache/ohlcv/SPY_1m_2018-05-01_2024-12-31_raw.parquet")
OUT_TRADES = Path("docs/research/post-w2/d3_mr.csv")
OUT_SUMMARY_CSV = Path("docs/research/post-w2/d3_mr_summary.csv")
OUT_SUMMARY_MD = Path("docs/research/post-w2/d3_mr_summary.md")

ET = ZoneInfo("America/New_York")

PURE_OOS_START = date(2024, 4, 1)


def cumulative_vwap(day_bars: pd.DataFrame) -> pd.Series:
    """Cumulative VWAP of typical price weighted by volume."""
    tp = (day_bars["high"] + day_bars["low"] + day_bars["close"]) / 3.0
    vol = day_bars["volume"].astype(float)
    num = (tp * vol).cumsum()
    den = vol.cumsum().replace(0, np.nan)
    return num / den


def run_day(day_bars: pd.DataFrame, trade_date: date) -> dict | None:
    """Run one day of the fade-open strategy. Return trade dict or None."""
    if len(day_bars) < 390:
        return None
    open_09_30 = day_bars.iloc[0]["open"]
    close_09_34 = day_bars.iloc[4]["close"]
    if close_09_34 == open_09_30:
        return None
    direction = "short" if close_09_34 > open_09_30 else "long"
    entry_price = float(day_bars.iloc[5]["close"])
    entry_time = day_bars.index[5]

    vwap = cumulative_vwap(day_bars)

    # Scan from bar 6 to 15:54 ET for VWAP cross
    exit_price = None
    exit_time = None
    outcome = None
    for i in range(6, 385):  # stop before last 5 bars
        p = float(day_bars.iloc[i]["close"])
        v = float(vwap.iloc[i])
        if direction == "long" and p >= v:
            exit_price, exit_time, outcome = p, day_bars.index[i], "vwap_cross"
            break
        if direction == "short" and p <= v:
            exit_price, exit_time, outcome = p, day_bars.index[i], "vwap_cross"
            break

    if exit_price is None:
        # 15:55 ET EOD flat
        exit_price = float(day_bars.iloc[385]["close"])
        exit_time = day_bars.index[385]
        outcome = "eod"

    pnl = (exit_price - entry_price) if direction == "long" else (entry_price - exit_price)
    return {
        "date": trade_date.isoformat(),
        "direction": direction,
        "entry_time": entry_time,
        "entry_price": entry_price,
        "exit_time": exit_time,
        "exit_price": exit_price,
        "outcome": outcome,
        "pnl": pnl,
    }


def _window_summary(df: pd.DataFrame, label: str) -> dict:
    if df.empty:
        return {"window": label, "n": 0, "sharpe_gross": float("nan"),
                "sortino_gross": float("nan"), "mdd": 0.0, "pf": float("nan"), "win_rate": 0.0}
    pnl = df["pnl"].astype(float)
    # Daily returns from per-trade pnl (one trade per day max by design)
    daily = pnl.reset_index(drop=True)
    sharpe = float(annualized_sharpe(daily)) if len(daily) > 5 else float("nan")
    wins = (pnl > 0).sum()
    losses = (pnl <= 0).sum()
```

```python
"""D4 — Overnight gap decomposition + naive gap-fill baseline (spec §2 D4).

Baseline rules:
  - |gap_pct| < 0.1% → skip (no-gap day)
  - up-gap: short at 09:30 open, target = prev_close, EOD flat, no stop
  - down-gap: mirror long

Decision gate: row 4 — pure_oos sharpe_gross ≥ 1.0.

Usage:
    .venv/bin/python scripts/diagnose_d4_gap_decomposition.py
"""

from __future__ import annotations

from datetime import date
from pathlib import Path

import numpy as np
import pandas as pd

from daytrader.research.bakeoff.metrics import annualized_sharpe

SPY_DAILY_CACHE = Path("data/cache/ohlcv/SPY_1d_2018-05-01_2024-12-31_raw.parquet")
OUT_GAP = Path("docs/research/post-w2/d4_gap.csv")
OUT_SUMMARY_CSV = Path("docs/research/post-w2/d4_gap_summary.csv")
OUT_SUMMARY_MD = Path("docs/research/post-w2/d4_gap_summary.md")

GAP_THRESHOLD = 0.001
PURE_OOS_START = date(2024, 4, 1)


def classify_gap(prev_close: float, open_price: float) -> str:
    gap_pct = (open_price - prev_close) / prev_close
    if gap_pct > GAP_THRESHOLD:
        return "up"
    if gap_pct < -GAP_THRESHOLD:
        return "down"
    return "none"


def evaluate_gap_fill(
    prev_close: float,
    open_price: float,
    intraday_high: float,
    intraday_low: float,
    close_price: float,
) -> dict | None:
    cls = classify_gap(prev_close, open_price)
    if cls == "none":
        return None

    if cls == "up":
        # Short the gap; target = prev_close (lower); EOD flat
        direction = "short"
        entry = open_price
        if intraday_low <= prev_close:
            exit_price, outcome = prev_close, "target"
        else:
            exit_price, outcome = close_price, "eod"
        pnl = entry - exit_price
    else:
        direction = "long"
        entry = open_price
        if intraday_high >= prev_close:
            exit_price, outcome = prev_close, "target"
        else:
            exit_price, outcome = close_price, "eod"
        pnl = exit_price - entry

    return {
        "direction": direction, "entry_price": entry,
        "exit_price": exit_price, "outcome": outcome, "pnl": pnl,
        "gap_class": cls,
    }


def _window_summary(df: pd.DataFrame, label: str) -> dict:
    if df.empty:
        return {"window": label, "n": 0, "sharpe_gross": float("nan"),
                "mdd": 0.0, "pf": float("nan"), "win_rate": 0.0}
    pnl = df["pnl"].astype(float)
    daily = pnl.reset_index(drop=True)
    sharpe = float(annualized_sharpe(daily)) if len(daily) > 5 else float("nan")
    wins = (pnl > 0).sum()
    losses = (pnl <= 0).sum()
    pf_denom = -pnl[pnl < 0].sum()
    pf = float(pnl[pnl > 0].sum() / pf_denom) if pf_denom > 0 else float("nan")
    eq = pnl.cumsum()
    mdd = float((eq - eq.cummax()).min()) if len(eq) else 0.0
    return {"window": label, "n": int(len(pnl)), "sharpe_gross": sharpe,
            "mdd": mdd, "pf": pf,
            "win_rate": float(wins / (wins + losses)) if (wins + losses) else 0.0}


def main():
    import sys
    if not SPY_DAILY_CACHE.exists():
        print(f"ERROR: {SPY_DAILY_CACHE} missing", file=sys.stderr)
        sys.exit(1)
    d = pd.read_parquet(SPY_DAILY_CACHE)
    if isinstance(d.columns, pd.MultiIndex):
        d = d["SPY"]
    d = d.sort_index()
    d["prev_close"] = d["close"].shift(1)
    d = d.dropna(subset=["prev_close"])

    trades = []
    for dt, row in d.iterrows():
        t = evaluate_gap_fill(
            prev_close=float(row["prev_close"]),
            open_price=float(row["open"]),
            intraday_high=float(row["high"]),
            intraday_low=float(row["low"]),
            close_price=float(row["close"]),
        )
        if t is not None:
            t["date"] = pd.Timestamp(dt).date().isoformat()
            trades.append(t)

    tr_df = pd.DataFrame(trades)
    OUT_GAP.parent.mkdir(parents=True, exist_ok=True)
    tr_df.to_csv(OUT_GAP, index=False)

    tr_df["trade_date"] = pd.to_datetime(tr_df["date"]).dt.date
    full = _window_summary(tr_df, "full_window")
    pure = tr_df[tr_df["trade_date"] >= PURE_OOS_START]
```

```python
"""D5 — Multi-instrument S1 sampling on QQQ + IWM (spec §2 D5).

Inline Databento loader for QQQ + IWM 1m ARCX.PILLAR over pure OOS only

Usage:

from __future__ import annotations

import os
import sys
from dataclasses import asdict
from datetime import date
from pathlib import Path

import pandas as pd

from daytrader.research.bakeoff.data import data_quality_report, filter_rth
from daytrader.research.bakeoff.metrics import annualized_sharpe
from daytrader.research.bakeoff.strategies.s1_orb import (
    S1a_ORB_TargetAndEOD,
    S1b_ORB_EODOnly,
)

CACHE_DIR = Path("data/cache/ohlcv")
OUT_DIR = Path("docs/research/post-w2")
SYMBOLS = ("QQQ", "IWM")
OOS_START = date(2024, 4, 1)
OOS_END = date(2024, 12, 31)

COST_CEILING = 3.0   # per spec §2 D5


def _cache_path(symbol: str, start: date, end: date) -> Path:
    return CACHE_DIR / f"{symbol}_1m_{start.isoformat()}_{end.isoformat()}_raw.parquet"


def _preflight(api_key: str) -> float:
    """Total cost estimate for both symbols, full OOS window."""
    import databento
    client = databento.Historical(api_key)
    total = 0.0
    for sym in SYMBOLS:
        c = client.metadata.get_cost(
            dataset="ARCX.PILLAR", schema="ohlcv-1m",
            symbols=[sym], stype_in="raw_symbol",
            start=OOS_START.isoformat(), end=OOS_END.isoformat(),
        )
        total += float(c)
    return total


def _fetch(symbol: str, api_key: str) -> pd.DataFrame:
    p = _cache_path(symbol, OOS_START, OOS_END)
    if p.exists():
        return pd.read_parquet(p)
    import databento
    client = databento.Historical(api_key)
    req = client.timeseries.get_range(
        dataset="ARCX.PILLAR", schema="ohlcv-1m",
        symbols=[symbol], stype_in="raw_symbol",
        start=OOS_START.isoformat(),
        end=(pd.Timestamp(OOS_END) + pd.Timedelta(days=1)).date().isoformat(),
    )
    df = req.to_df()
    if df.index.tz is None:
        df.index = df.index.tz_localize("UTC")
    df.to_parquet(p)
    return df


def _run_one(bars, symbol, variant_name, cls):
    strat = cls(symbol=symbol, point_value=1.0)  # ETFs: $1/point
    trades = strat.run(bars)
    tr_df = pd.DataFrame([{**asdict(t), "outcome": t.outcome.value} for t in trades])
    tr_df.to_csv(OUT_DIR / f"d5_{symbol.lower()}_{variant_name}.csv", index=False)

    if not trades:
        return {"symbol": symbol, "variant": variant_name, "n": 0, "sharpe_gross": float("nan")}
    pnl = pd.Series([
        (t.exit_price - t.entry_price) if t.direction == "long" else (t.entry_price - t.exit_price)
        for t in trades
    ])
    sharpe = float(annualized_sharpe(pnl)) if len(pnl) > 5 else float("nan")
    return {"symbol": symbol, "variant": variant_name, "n": len(trades), "sharpe_gross": sharpe}


def build_compare_table(rows: list[dict]) -> pd.DataFrame:
    return pd.DataFrame(rows, columns=["symbol", "variant", "n", "sharpe_gross"])


def main():

    if not api_key:
        print("ERROR: DATABENTO_API_KEY not set", file=sys.stderr)
        sys.exit(1)

    if cost > COST_CEILING:
        print(f"ABORT: cost > ${COST_CEILING}. Row 3 marked deferred.", file=sys.stderr)
        sys.exit(2)

        from zoneinfo import ZoneInfo
        et = ZoneInfo("America/New_York")
        if low:
            mask = ~pd.Series(rth.index.tz_convert(et).date, index=rth.index).isin(set(low))
            rth = rth[mask]

        for vname, cls in [("s1a", S1a_ORB_TargetAndEOD), ("s1b", S1b_ORB_EODOnly)]:
            results.append(_run_one(rth, sym, vname, cls))

    compare = build_compare_table(results)

    md = ["# D5 Multi-Instrument S1 Sampling — QQQ + IWM", "",
          f"Pure OOS only: {OOS_START} → {OOS_END}", "",
          compare.to_markdown(index=False, floatfmt=".4f")]

    print("\n".join(md))

    main()
```

```python
    sharpe = float(annualized_sharpe(pnl.reset_index(drop=True))) if len(pnl) > 5 else float("nan")
    wins = (pnl > 0).sum()
    losses = (pnl <= 0).sum()
    return {"partition": label, "n": int(len(pnl)), "sharpe_gross": sharpe,
            "win_rate": float(wins / (wins + losses)) if (wins + losses) else 0.0,
            "mean_pnl": float(pnl.mean())}


def _load_trades() -> pd.DataFrame:
    import sys
    if not (PLAN3_S1A.exists() and PLAN3_S1B.exists()):
        print(f"ERROR: {PLAN3_S1A} or {PLAN3_S1B} missing", file=sys.stderr)
        sys.exit(1)
    a = pd.read_csv(PLAN3_S1A); a["variant"] = "s1a"
    b = pd.read_csv(PLAN3_S1B); b["variant"] = "s1b"
    t = pd.concat([a, b], ignore_index=True)
    t["trade_date"] = pd.to_datetime(t["date"]).dt.date
    t["pnl"] = ((t["exit_price"] - t["entry_price"]).where(
        t["direction"] == "long",
        t["entry_price"] - t["exit_price"]
    ))
    return t


def main():
    trades = _load_trades()
    events = pd.read_csv(EVENT_CAL)
    event_dates = set(events["date"].astype(str))

    trades["is_event_day"] = trades["trade_date"].astype(str).isin(event_dates)
    OUT_EVENTS.parent.mkdir(parents=True, exist_ok=True)
    trades.to_csv(OUT_EVENTS, index=False)

    on, off = partition_trades(trades, event_dates)
    summary = pd.DataFrame([
        summarize_partition(on, "on_event"),
        summarize_partition(off, "off_event"),
    ])
    summary.to_csv(OUT_SUMMARY_CSV, index=False)

    md = ["# D6 Event-Day Isolation — S1 trade log partitioning", "",
          "Using 2024 FOMC/CPI/NFP calendar.",
          "",
          summary.to_markdown(index=False, floatfmt=".4f"),
          "",
          "**Decision rule (§3.1 row 5):** `on_event` row requires both "
          "`sharpe_gross ≥ 1.0` AND `n ≥ 30`."]
    OUT_SUMMARY_MD.write_text("\n".join(md) + "\n")
    print(OUT_SUMMARY_MD.read_text())


if __name__ == "__main__":
    main()
```
